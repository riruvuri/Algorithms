1. In N choose K combinations, why do we have exclude and include options?
Given two integers n and k, find all the possible unique combinations of k numbers in range 1 to n.


static void helper(ArrayList<ArrayList<Integer>> resultList, ArrayList<Integer> result,
            Integer k, int[] nums, int index){

            if(result.size() == k){
                resultList.add(new ArrayList<>(result));
                return;
            }


            if(index == nums.length) return;


            // exclude
            helper(resultList, result, k , nums, index+1);

            // include
            result.add(nums[index]);
            helper(resultList, result, k , nums, index+1);
            result.remove(result.size()-1);
    }
